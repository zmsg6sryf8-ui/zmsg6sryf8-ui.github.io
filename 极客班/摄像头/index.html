    cv2.waitKey(0)

    等待键盘输入，参数0表示无限等待，其他数字表示等待毫秒数

    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)

    将彩色图像转换为灰度图像

    edges = cv2.Canny(gray, 25, 100)

    初始化摄像头设备

    创建VideoCapture对象，参数0表示默认摄像头

    Camera = cv2.VideoCapture(0)

    while True：
   从摄像头读取一帧图像
   ret, frame = Camera.read()  # ret返回读取成功与否的布尔值，frame返回捕获的图像帧
  灰度化处理：将彩色图像转换为灰度图像

 边缘化处理：使用Canny算法进行边缘检测

    # 显示原始图像
   cv2.imshow("Original Camera", frame) # 显示原始的彩色图像
    # 显示灰度化处理后的图像

    # 显示边缘检测处理后的图像
   cv2.imshow("Edge Detection", edges_frame) # 显示边缘检测结果

   cv2.waitKey(1)  # 等待1秒后继续执行，期间可接收键盘输入

    人脸识别

    import cv2

    打开默认摄像头（通常为电脑内置摄像头）

    Camera = cv2.VideoCapture(0)

    加载OpenCV自带的人脸检测级联分类器（Haar特征分类器）

    face = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")

    创建主循环，持续处理视频流

    while True:
从摄像头读取一帧图像，ret表示读取是否成功，frame是图像数据
    ret, frame = Camera.read()

    # 使用人脸检测器检测当前帧中的人脸，返回人脸位置的矩形坐标列表
    FD = face.detectMultiScale(frame)

    # 遍历检测到的每个人脸位置
    for (x, y, w, h) in FD:
        # 在检测到的人脸周围绘制绿色矩形框，线宽为2像素
        cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 0), 2)

    # 显示带有人脸框的视频帧
    cv2.imshow("frame", frame)

    # 等待1毫秒并检查是否按下'q'键，如果是则退出循环
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

    释放摄像头资源

    Camera.release()

    关闭所有OpenCV创建的窗口

    cv2.destroyAllWindows()

    3识别图像

    import cv2

    打开默认摄像头

    Camera = cv2.VideoCapture(0)

    主循环，持续处理视频流

    while True:

    # 从摄像头读取一帧图像，ret表示读取是否成功，frame是图像数据

   ret, frame = Camera.read()

    # 将彩色图像转换为灰度图像，便于边缘检测处理
   gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)

    # 使用Canny算法检测图像中的边缘，50和150分别是低阈值和高阈值
   edges = cv2.Canny(gray, 50, 150)

    # 在边缘图像中查找轮廓，RETR_EXTERNAL表示只检测最外层轮廓
    # CHAIN_APPROX_SIMPLE表示压缩水平、垂直和对角线段，只保留它们的端点
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

    # 遍历所有检测到的轮廓
    for i in contours:

        # 使用多边形近似轮廓，epsilon参数控制近似精度（原始轮廓周长的一定比例）
        # 第二个True表示轮廓是闭合的
        epsilon = 0.02 * cv2.arcLength(i, True)
        # 使用Douglas-Peucker算法对轮廓进行多边形近似
        approx = cv2.approxPolyDP(i, epsilon, True)
        # 获取近似多边形的外接矩形坐标和尺寸
        x, y, w, h = cv2.boundingRect(approx)
        # 判断轮廓的顶点数量来确定形状类型
        if len(approx) == 3:
            # 如果是三角形，绘制红色三角形框（使用多边形绘制而不是矩形）
            cv2.polylines(frame, [approx], True, (0, 0, 255), 2)
            print("三角形")  # 在控制台输出形状名称
        elif len(approx) == 4:
            # 如果是矩形，绘制红色矩形框
            cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 0, 255), 2)
            print("矩形")  # 在控制台输出形状名称
    # 显示处理后的图像帧
    cv2.imshow("frame", frame)

    # 等待1毫秒，如果按下'q'键则退出循环
    if cv2.waitKey(1) & 0xFF == ord('q'):
       break

    释放摄像头资源

    Camera.release()

    关闭所有OpenCV创建的窗口

    cv2.destroyAllWindows()
