    import cv2

    打开默认摄像头（通常为电脑内置摄像头）

    Camera = cv2.VideoCapture(0)

    加载OpenCV自带的人脸检测级联分类器（Haar特征分类器）

    face = cv2.CascadeClassifier(cv2.data.haarcascades + "haarcascade_frontalface_default.xml")

    创建主循环，持续处理视频流

    while True: # 从摄像头读取一帧图像，ret表示读取是否成功，frame是图像数据 ret, frame = Camera.read()

    # 使用人脸检测器检测当前帧中的人脸，返回人脸位置的矩形坐标列表
FD = face.detectMultiScale(frame)

# 遍历检测到的每个人脸位置
for (x, y, w, h) in FD:
    # 在检测到的人脸周围绘制绿色矩形框，线宽为2像素
    cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 0), 2)

# 显示带有人脸框的视频帧
cv2.imshow("frame", frame)

# 等待1毫秒并检查是否按下'q'键，如果是则退出循环
if cv2.waitKey(1) & 0xFF == ord('q'):
    break

    释放摄像头资源

    Camera.release()

    关闭所有OpenCV创建的窗口

    cv2.destroyAllWindows()
